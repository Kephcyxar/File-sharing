  test("UM_010 - User Management: Department sort cycles", async () => {
    // Go to a different page and back to get a clean table render
    await page.getByRole("link", { name: "VM Management" }).click()
    await page.waitForSelector("tbody tr", { timeout: 8000 })
    await page.getByRole("link", { name: "User Management" }).click()
    await page.waitForSelector("tbody tr", { timeout: 8000 })

    // Clear search box if anything was left from previous tests
    const search = page.getByTestId("user-search-input")
    await search.fill("")
    await page.keyboard.press("Enter").catch(() => {})
    await page.waitForTimeout(150)

    // ---- API: fetch all users, extract and sort departments
    const api = await request.newContext({ ignoreHTTPSErrors: true })
    const resp = await api.get("/api/user/getAllUser/")
    expect(resp.ok()).toBeTruthy()
    const data = await resp.json()
    await api.dispose()

    const usersArr = Array.isArray(data) ? data : Array.isArray(data?.users) ? data.users : []
    const rawDepartments = usersArr.map((u) => u?.department).filter(Boolean)

    const normalizeDeptForUI = (s) => {
      if (!s) return s
      const t = String(s).trim()
      return t.includes("-") ? t.split("-").slice(1).join("-").trim() : t
    }

    const departmentsNormalized = Array.from(new Set(rawDepartments.map(normalizeDeptForUI).filter(Boolean)))

    const collate = (a, b) => a.localeCompare(b, undefined, { sensitivity: "accent" })

    const sortedAsc = [...departmentsNormalized].sort(collate)
    expect(sortedAsc.length).toBeGreaterThan(0)

    const headers = page.locator("thead tr th")

    async function getColumnIndex(headerText) {
      const count = await headers.count()
      for (let i = 0; i < count; i++) {
        const t = (await headers.nth(i).innerText()).trim()
        if (new RegExp(`^${headerText}$`, "i").test(t)) return i
      }
      throw new Error(`${headerText} column not found`)
    }

    async function getFirstCellText(colIdx) {
      const firstRow = page.locator("tbody tr").first()
      await expect(firstRow).toBeVisible()
      return (await firstRow.locator("td").nth(colIdx).innerText()).trim()
    }

    async function getFirstRowDept() {
      const depIdx = await getColumnIndex("Department")
      return getFirstCellText(depIdx)
    }

    async function getFirstRowUsername() {
      const uIdx = await getColumnIndex("Username")
      return getFirstCellText(uIdx)
    }

    // Store the original first Username to validate the "reset" state
    const originalFirstUsername = await getFirstRowUsername()
    expect(originalFirstUsername.length).toBeGreaterThan(0)

    // Locate the Department header + sort icon
    const deptHeader = page.locator('th:has(.header-cell .header-text:has-text("Department"))').first()
    await expect(deptHeader).toBeVisible()
    const sortIcon = deptHeader.locator(".header-icons svg").first()

    async function clickSort() {
      if (await sortIcon.count()) {
        await sortIcon.click({ force: true })
      } else {
        await deptHeader.locator(".header-cell").click({ force: true })
      }
      await page.waitForTimeout(200)
    }

    // 1) ASC: first row's Department should equal the first item of the sorted list
    await clickSort()
    const firstDeptAsc = normalizeDeptForUI(await getFirstRowDept())
    expect(firstDeptAsc).toBe(sortedAsc[0])

    // 2) DESC: first row's Department should equal the last item of the sorted list
    await clickSort()
    const firstDeptDesc = normalizeDeptForUI(await getFirstRowDept())
    expect(firstDeptDesc).toBe(sortedAsc[sortedAsc.length - 1])

    // 3) RESET: first row's Username should match the one we stored at the beginning
    await clickSort()
    const firstUsernameAfterReset = await getFirstRowUsername()
    expect(firstUsernameAfterReset).toBe(originalFirstUsername)
  })

  test("UM_011 - Filter Department via header filter", async () => {
    // Navigate away & back to force a clean render
    await page.getByRole("link", { name: "VM Management" }).click()
    await page.waitForSelector("tbody tr", { timeout: 8000 })
    await page.getByRole("link", { name: "User Management" }).click()
    await page.waitForSelector("tbody tr", { timeout: 8000 })

    // Make sure table header is visible at the top of the viewport
    await page.evaluate(() => window.scrollTo(0, 0))

    const deptHeader = page.locator('th:has(.header-cell .header-text:has-text("Department"))').first()
    await expect(deptHeader).toBeVisible()
    await deptHeader.scrollIntoViewIfNeeded()

    const filterIcon = deptHeader.locator(".header-icons .filter-icon").first()
    await expect(filterIcon).toBeVisible()
    await filterIcon.click({ force: true })

    // AntD dropdown / popover that contains our filter UI
    const popup = page
      .locator(".custom-filter-dropdown, .modal-filter-dropdown, .ant-dropdown, .ant-popover")
      .filter({ has: page.locator(".filter-dropdown") })
      .last()
    await popup.waitFor({ state: "visible" })

    // Scroll inside the filter pane, not the page
    const pane = popup.locator(".filter-options").first()
    await expect(pane).toBeVisible()
    await pane.evaluate((el) => {
      el.scrollTop = 0
    })

    // Prefer selecting the INPUT directly
    const firstInput = pane.locator('input[type="checkbox"]').first()
    await expect(firstInput).toBeVisible()

    // Capture the visible label text for assertion
    const cbId = await firstInput.getAttribute("id")
    const firstLabel = pane.locator(`label[for="${cbId}"]`).first()
    const labelText = (await firstLabel.innerText()).trim()

    // Check the checkbox (no click on label)
    await page.waitForTimeout(1000)
    await firstInput.check({ force: true })
    await page.waitForTimeout(2000)
    await expect(firstInput).toBeChecked()

    // Close the popup so the filter applies
    await page.keyboard.press("Escape").catch(() => {})
    await page.waitForTimeout(200)

    // Find the "Department" column index
    const headers = page.locator("thead tr th")
    const hCount = await headers.count()
    let depIdx = -1
    for (let i = 0; i < hCount; i++) {
      const t = (await headers.nth(i).innerText()).trim()
      if (/^department$/i.test(t)) {
        depIdx = i
        break
      }
    }
    expect(depIdx).toBeGreaterThanOrEqual(0)

    // Assert all visible rows match the chosen department
    const rows = page.locator("tbody tr")
    const rCount = await rows.count()
    expect(rCount).toBeGreaterThan(0)

    for (let i = 0; i < rCount; i++) {
      const depCell = rows.nth(i).locator("td").nth(depIdx)
      await expect(depCell).toContainText(labelText)
    }

    await filterIcon.click()
    const popupReset = page
      .locator(".custom-filter-dropdown, .modal-filter-dropdown, .ant-dropdown, .ant-popover")
      .filter({ has: page.locator(".filter-dropdown") })
      .last()

    await popupReset.waitFor({ state: "visible" })

    const paneReset = popupReset.locator(".filter-options").first()
    const firstInputReset = paneReset.locator('input[type="checkbox"]').first()

    await firstInputReset.uncheck({ force: true })

    await page.keyboard.press("Escape")
  })

    await search.fill(ldapUsername)
    await search.press("Enter").catch(() => {})
    const existingRow = page.locator("tbody tr", { hasText: ldapUsername }).first()
    const rowCount = await existingRow.count()

    if (rowCount > 0) {
      loggerFormat("Existing user found. Extracting user_id...")
      const userId = (await existingRow.locator("td").nth(1).innerText()).trim()
      loggerFormat(`Deleting user with ID: ${userId}`)
      await page.getByTestId(`delete-user-button-${userId}`).click()
      await page.waitForSelector('[data-testid="confirm-message-modal"]', { state: "visible" })
      await page.getByRole("button", { name: "Confirm" }).click()
      await expect(page.locator(`tr:has-text("${ldapUsername}")`)).toHaveCount(0)
      loggerFormat("Pre-existing user deleted.")
    } else {
      loggerFormat("No existing user found. Continuing test.")
    }

    await search.fill("")
    await search.press("Enter").catch(() => {})

    // Step 1 — Add LDAP user
    await page.getByTestId("add-user-button").click()
    const modal = page.getByRole("dialog")
    await expect(modal).toBeVisible()
    await modal.getByTestId("add-user-search-input").fill(ldapUsername)
    await page.waitForTimeout(1000)
    const ldapCheckbox = modal.getByTestId(`add-user-checkbox-${ldapUsername}`)
    await ldapCheckbox.check()
    await modal.getByRole("button", { name: "Add Selected" }).click()
    await expect(page.getByRole("listitem").filter({ hasText: ldapUsername })).toBeVisible()
    await page.getByRole("button", { name: "Confirm" }).click()
    loggerFormat(`LDAP user ${ldapUsername} added successfully.`)

    // Step 2 — Try manual create with same username
    await page.getByTestId("add-user-button").click()
    const modal2 = page.getByRole("dialog")
    await expect(modal2).toBeVisible()
    await modal2.getByTestId("manual-tab").click()
    await modal2.getByTestId("manual-username-input").fill(ldapUsername)
    await modal2.getByTestId("manual-title-input").fill("QA Tester")
    await modal2.getByTestId("manual-department-input").fill("Testing Department")
    await modal2.getByTestId("manual-password-input").fill("Test1234")
    await modal2.getByTestId("manual-confirm-password-input").fill("Test1234")
    await modal2.getByTestId("create-user-button").click()

    // Expected alert
    await expect(page.getByText(/Username already exists/i)).toBeVisible()
    loggerFormat("Duplicate username prevented correctly.")
    await page.getByRole("button", { name: "Cancel" }).click()

    // Step 3 — Delete the LDAP user from table
    loggerFormat("Deleting LDAP user from table")
    await page.getByTestId("user-search-input").fill(ldapUsername)
    await page.getByTestId(`delete-user-button-${new_user_id}`).click()
    await page.getByTestId("confirm-message-modal").waitFor({ state: "visible" })
    await page.getByRole("button", { name: "Confirm" }).click()
    await expect(page.locator(`tr:has-text("${ldapUsername}")`)).toHaveCount(0)
    loggerFormat("LDAP user deleted successfully.")

    // Step 4 — Create manual user with same username
    await page.getByTestId("add-user-button").click()
    const modal3 = page.getByRole("dialog")
    await expect(modal3).toBeVisible()
    await modal3.getByTestId("manual-tab").click()
    await modal3.getByTestId("manual-username-input").fill(ldapUsername)
    await modal3.getByTestId("manual-title-input").fill("QA Tester")
    await modal3.getByTestId("manual-department-input").fill("Testing Department")
    await modal3.getByTestId("manual-password-input").fill("Test1234")
    await modal3.getByTestId("manual-confirm-password-input").fill("Test1234")
    await modal3.getByTestId("create-user-button").click()
    await expect(modal3).toBeHidden()
    loggerFormat(`Manual user ${ldapUsername} created successfully.`)

    // Step 5 — Verify LDAP list hides this username
    await page.getByTestId("add-user-button").click()
    const modal4 = page.getByRole("dialog")
    await expect(modal4).toBeVisible()
    await modal4.getByTestId("add-user-search-input").fill(ldapUsername)
    const ldapRow = modal4.locator(`tr:has-text("${ldapUsername}")`)
    await expect(ldapRow).toHaveCount(0)
    loggerFormat("Manual created username correctly hidden from LDAP list.")
    await page.getByRole("button", { name: "Cancel" }).click()

    // Step 6 - Remove back the user
    loggerFormat("Checking if test user already exists in table")

    await search.fill(ldapUsername)
    await search.press("Enter").catch(() => {})

    if (rowCount > 0) {
      loggerFormat("Existing user found. Extracting user_id...")
      const userId = (await existingRow.locator("td").nth(1).innerText()).trim()
      loggerFormat(`Deleting user with ID: ${userId}`)
      await page.getByTestId(`delete-user-button-${userId}`).click()
      await page.waitForSelector('[data-testid="confirm-message-modal"]', { state: "visible" })
      await page.getByRole("button", { name: "Confirm" }).click()
      await expect(page.locator(`tr:has-text("${ldapUsername}")`)).toHaveCount(0)
      loggerFormat("Pre-existing user deleted.")
    } else {
      loggerFormat("No existing user found. Continuing test.")
    }

    await search.fill("")
    await search.press("Enter").catch(() => {})
  })
